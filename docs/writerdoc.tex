\section{ngames/search/writer.pl}

\label{sec:writer}

\begin{description}
    \predicate[det]{copy_default}{1}{+DefaultsPath:str}
Copy the action situation description files containing the default
configuration into the current directory. The copied files are modified, not
the original ones.

\begin{arguments}
\arg{DefaultsPath} & Path to folder with default action situation description. \\
\end{arguments}

    \predicate[det]{copy_default_file}{2}{+FromFile:str, +ToFile:str}
Auxiliary predicate to copy a single file.

\begin{arguments}
\arg{FromFile} & Path to file to be copied. \\
\arg{ToFile} & Path where the copy is to be stored. \\
\end{arguments}

    \predicate[det]{update_action_situation_description}{4}{+NormRepository:str, +Defaults:str, +NormativeSystem:dict, +ID:str}
Update a default action situation description with the additional norms
(and their respective priorities) of a normative system.

\begin{arguments}
\arg{NormRepository} & Path to the repository of additional norms. \\
\arg{Default} & Path to the files that contain the default description. \\
\arg{NormativeSystem} & Dictionary of name keys and strictly positive integer
values, that correspond to the priorities that their
rules will be assigned. \\
\arg{ID} & Action situation identifier. \\
\end{arguments}

    \predicate[det]{process_norm_stream}{3}{+InStream:stream, +NormativeSystem:dict, +ID:str}
Process the repository of additional norms and add the corresponding rules
and other predicates for the norms that are included in the normative system.

\begin{arguments}
\arg{InStream} & Norm repository input stream. \\
\arg{NormativeSystem} & Dictionary of name keys and strictly positive integer
values, that correspond to the priorities that their
rules will be assigned. \\
\arg{ID} & Action situation identifier. \\
\end{arguments}

    \predicate[det]{process_norm_dict}{3}{+Norm:dict, +Priority:int, ID:str}
Add a norm to the action situation description file. Its '\predref{rule}{4}`
statements are assigned priotity given by '\arg{Priority}`.

\begin{arguments}
\arg{Norm} & A dictionary with all the information related to the norm.
Retrieved from the norm repository. \\
\arg{Priority} & Strictly positive integer. \\
\arg{ID} & Action situation identifier. \\
\end{arguments}

    \predicate[det]{add_predicates}{2}{+L:list, +Stream:stream}
Add the predicates as dynamic to the corresponding file of the action
situation description.

\begin{arguments}
\arg{L} & List of predicate/arity to be included as dynamic. \\
\arg{Stream} & Output file stream. \\
\end{arguments}

    \predicate[det]{add_incompatibles}{2}{+L:list, +Stream:stream}
Add the predicates as incompatible to the corresponding file of the action
situation description.

\begin{arguments}
\arg{L} & List of predicates to be included as incompatible. \\
\arg{Stream} & Output file stream. \\
\end{arguments}

    \predicate[det]{add_initials}{2}{+L:list, +Stream:stream}
Add the predicates as initial conditions to the corresponding file of the
action situation description.

\begin{arguments}
\arg{L} & List of predicates to be included as initial conditions. \\
\arg{Stream} & Output file stream. \\
\end{arguments}

    \predicate[det]{add_rules}{5}{+L:list, +ID:str, +Type:str, +Priority:int, +Stream:stream}
Add the rules to the action situation description, with the input
priority and rule type.

\begin{arguments}
\arg{L} & List of rule content arguments to be added. \\
\arg{ID} & Action situation identifier. \\
\arg{Type} & Rule type. One of either 'boundary`, 'position`, 'choice`,
'control` or 'payoff`. \\
\arg{Priority} & Strictly positive integer. \\
\arg{Stream} & Output file stream. \\
\end{arguments}

    \predicate[det]{build_ASL_description}{4}{+DefaultsPath:str, +NormsPath:str, +NormativeSystem:dict, +ID:str}
Make a copy of the default action situation description files and expand it
with the norms selected from a norm repository by the input normative system.

\begin{arguments}
\arg{DefaultsPath} & Path to the folder with the default action situation
description files. \\
\arg{NormsPath} & Path to the norm repository file. \\
\arg{NormativeSystem} & Dictionary mapping norms in the repository to a
strictly positive integer (aka the priority that
the norm has). \\
\arg{ID} & Action situation identifier. \\
\end{arguments}
\end{description}
